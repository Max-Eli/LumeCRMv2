# Production-only dependencies.
#
# Kept separate from `requirements.txt` so local dev (where the file-
# based email backend, console logging, and runserver are fine) doesn't
# pull in AWS / WhiteNoise / Sentry weight. The Dockerfile installs
# both files; local dev installs only `requirements.txt`.
#
# Pin every package — surprises from a transient minor-version bump
# during a Friday deploy are not on the menu.

# WSGI server. 21.x is the current stable line.
gunicorn==23.0.0

# Static-file serving from inside the gunicorn process (admin + DRF
# browsable API). CloudFront caches in front, but WhiteNoise needs to
# be the fallback.
whitenoise==6.7.0

# AWS SDK — used for SES (email), Secrets Manager, S3 signed URLs.
boto3==1.35.50

# AWS-specific Django integrations.
django-ses==4.1.1                    # SES email backend (ADR 0012)
django-storages[s3]==1.14.4          # S3 default storage for media files

# JSON-structured logs so CloudWatch Insights queries don't have to
# parse free-form text. PHI-scrubbing filter is in lume_crm/logging.py.
python-json-logger==2.0.7

# MFA — TOTP via django-otp. Phase 0c v1 ships with this; Cognito
# migration deferred (see PROJECT_PLAN.md §4.5).
django-otp==1.5.4
qrcode[pil]==7.4.2                   # QR rendering for TOTP enrollment

# In-process rate limiting for login endpoints. (Fargate is multi-
# instance, so this is best-effort; for hardened limits we use ALB
# WAF rules. django-ratelimit is still useful as a defense layer.)
django-ratelimit==4.1.0
